Centos搭建Hadoop集群

2019-12-13 Comments

创建三台虚拟机编辑主机名为
修改主机名

hostnamectl set-hostname master

master
slaver1
slaver2

一、配置机器

  1. 配置master IP
    #vi /etc/sysconfig/network-scripts/ifcfg-eno16777736
    BOOTPROTO=static
    …..
    ONBOOT=yes
    IPADDR=192.168.200.101
    GATEWAY=192.168.200.1
    PREFIX=24

配置slaver1 IP
#vi /etc/sysconfig/network-scripts/ifcfg-eno16777736
BOOTPROTO=static
…..
ONBOOT=yes
IPADDR=192.168.200.102
GATEWAY=192.168.200.1
PREFIX=24

配置slaver2 IP
#vi /etc/sysconfig/network-scripts/ifcfg-eno16777736
BOOTPROTO=static
…..
ONBOOT=yes
IPADDR=192.168.200.103
GATEWAY=192.168.200.1
PREFIX=24

2.重启网络（每台主机都要执行）
#systemctl restart network

3.关闭防火墙及安全策略（每台主机都要执行）
#systemctl stop firewalld

systemctl disable firewalld

setenforce 0

systemctl disable enforce

vi /etc/selinux/config

SELINUX=disabled

  1. 配置主机映射
    #vi /etc/hosts

  2. 0.0.1 localhost localhost.localdomain localhost4 localhost4.localdomain4
    ::1 localhost localhost.localdomain localhost6 localhost6.localdomain6

  3. 168.200.101 master

  4. 168.200.102 slaver1

  5. 168.200.103 slaver2

  6. 将主机映射发送到其他两台电脑
    scp -r /etc/hosts slaver1:/etc
    输入密码
    scp -r /etc/hosts slaver2:/etc
    输入密码
    二安装jdk

  7. 上传jdk包到 /opt下；

  8. 解压

  9. 配置环境变量
    #vi /etc/profile
    export JAVA_HOME=/opt/jdk1.8.0_77
    export PATH=$PATH:$JAVA_HOME/bin
    重启profile

    source /etc/profile

    检查

    java –version

    检查结果为：
    java version “1.8.0_77”
    Java(TM) SE Runtime Environment (build 1.8.0_77-b03)
    Java HotSpot(TM) 64-Bit Server VM (build 25.77-b03, mixed mode)

三、在master机上部署hadoop
1.上传压缩包
hadoop-2.7.7.tar.gz
2.解压

cd /opt

#tar -zxvf hadoop-2.7.7.tar.gz

mkdir hadoop

#mv hadoop-2.7.7/* /opt/hadoop/

  1. 修改配置文件

    cd /opt/hadoop/etc/hadoop

    (1)hadoop-env.sh

    vi hadoop-env.sh

    export JAVA_HOME=/opt/jdk1.8.0_77

(2)core-site.xml

vi core-site.xml

fs.defaultFS hdfs://master:9000 hadoop.tmp.dir /opt/hadoopdata/hadoop

(3)hdfs-site.xml
#vi hdfs-site.xml

dfs.replication
3

dfs.name.dir /opt/hadoopdata/hadoop/name

dfs.data.dir
/opt/hadoopdata/hadoop/data

 dfs.namenode.secondary.http-address slaver1:50090 dfs.permissions.enabled false

(4)mapred-site.xml

mv mapred-site.xml.template mapred-site.xml

vi mapred-site.xml

mapreduce.framework.name yarn

(5)yarn-site.xml

vi yarn-site.xml

yarn.nodemanager.aux-services mapreduce_shuffle yarn.resourcemanager.hostname slaver2

(6)slaves

vi slaves

master
slaver1
slaver2

  1. 创建hadoop用到的目录

    mkdir /opt/hadoopdata

    mkdir /opt/hadoopdata/data

    mkdir /opt/hadoopdata/name

  2. 配置profile

    vi /etc/profile

    export JAVA_HOME=/opt/jdk1.8.0_77
    export HADOOP_HOME=/opt/hadoop
    export PATH=$PATH:$JAVA_HOME/bin:$HADOOP_HOME/bin:$HADOOP_HOME/sbin

    source /etc/profile

    检查

    hadoop version

    检查结果为：
    Hadoop 2.7.7
    Subversion Unknown -r c1aad84bd27cd79c3d1a7dd58202a8c3ee1ed3ac
    Compiled by stevel on 2018-07-18T22:47Z
    Compiled with protoc 2.5.0
    From source with checksum 792e15d20b12c74bd6f19a1fb886490
    This command was run using /opt/hadoop/share/hadoop/common/hadoop-common-2.7.7.jar

五、配置免密码登录
##master
#cd
#cd .ssh
#ssh-keygen
一路回车
#ssh-copy-id master
#ssh-copy-id slaver1
#ssh-copy-id slaver2
检查：
#ssh master
#exit
#ssh slaver1
#exit
#ssh slaver2
#exit

六、安装时间同步服务器

##master机上

yum -y install ntp

vi /etc/ntp.conf

注释或者删除以下四行
server 0.centos.pool.ntp.org iburst
server 1.centos.pool.ntp.org iburst
server 2.centos.pool.ntp.org iburst
server 3.centos.pool.ntp.org iburst
添加以下两行
server 127.127.1.0
fudge 127.127.1.0 stratum 10
#systemctl enable ntpd
#systemctl start ntpd

slaver1机上

yum -y install ntpdate

ntpdate master

systemctl enable ntpdate

slaver2机上

yum -y install ntpdate

ntpdate master

systemctl enable ntpdate
